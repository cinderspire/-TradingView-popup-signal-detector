## üóÑÔ∏è Database Schema

### Users Table
```sql
users {
  id: UUID PRIMARY KEY
  email: VARCHAR(255) UNIQUE
  password_hash: VARCHAR(255)
  username: VARCHAR(50) UNIQUE
  full_name: VARCHAR(100)
  avatar_url: VARCHAR(255)
  role: ENUM('user', 'provider', 'admin')
  verified: BOOLEAN DEFAULT false
  created_at: TIMESTAMP
  updated_at: TIMESTAMP
}
```

### Strategies Table
```sql
strategies {
  id: UUID PRIMARY KEY
  provider_id: UUID FOREIGN KEY -> users.id
  name: VARCHAR(100)
  description: TEXT
  type: ENUM('dca', 'grid', 'scalping', 'swing', 'momentum', 'custom')
  category: ENUM('crypto', 'forex', 'stocks', 'commodities')
  price: DECIMAL(10,2)
  status: ENUM('active', 'paused', 'archived')
  verified: BOOLEAN DEFAULT false

  -- Performance Metrics
  total_roi: DECIMAL(10,2)
  sharpe_ratio: DECIMAL(5,2)
  sortino_ratio: DECIMAL(5,2)
  win_rate: DECIMAL(5,2)
  avg_roi_per_trade: DECIMAL(5,2)
  risk_reward_ratio: DECIMAL(5,2)
  max_drawdown: DECIMAL(5,2)
  total_trades: INTEGER
  profitable_trades: INTEGER
  profit_factor: DECIMAL(5,2)

  -- Settings
  min_investment: DECIMAL(10,2)
  max_investment: DECIMAL(10,2)
  supported_exchanges: JSON
  supported_pairs: JSON
  timeframes: JSON

  -- Metadata
  downloads: INTEGER DEFAULT 0
  rating: DECIMAL(3,2)
  reviews_count: INTEGER DEFAULT 0
  created_at: TIMESTAMP
  updated_at: TIMESTAMP
}
```

### Signal Providers Table
```sql
signal_providers {
  id: UUID PRIMARY KEY
  user_id: UUID FOREIGN KEY -> users.id
  display_name: VARCHAR(100)
  bio: TEXT

  years_experience: INTEGER

  -- Performance

  total_signals_sent: INTEGER DEFAULT 0
  win_rate: DECIMAL(5,2)
  total_roi: DECIMAL(10,2)
  avg_roi_per_signal: DECIMAL(5,2)
  sharpe_ratio: DECIMAL(5,2)
  sortino_ratio: DECIMAL(5,2)
  max_drawdown: DECIMAL(5,2)
  avg_latency_ms: INTEGER

  -- Pricing
  subscription_price: DECIMAL(10,2)

  -- Settings
  supported_exchanges: JSON
  active_pairs: JSON
  trading_hours: JSON

  rating: DECIMAL(3,2)
  created_at: TIMESTAMP
  updated_at: TIMESTAMP
}
```

### Signals Table
```sql
signals {
  id: UUID PRIMARY KEY
  provider_id: UUID FOREIGN KEY -> signal_providers.id
  symbol: VARCHAR(20)
  exchange: VARCHAR(50)
  type: ENUM('long', 'short')
  entry_price: DECIMAL(18,8)
  stop_loss: DECIMAL(18,8)

  leverage: INTEGER
  risk_reward: DECIMAL(5,2)

  -- Results
  exit_price: DECIMAL(18,8)
  pnl_percent: DECIMAL(10,2)
  duration_minutes: INTEGER

  -- Metadata
  signal_text: TEXT
  sent_at: TIMESTAMP
  closed_at: TIMESTAMP
  created_at: TIMESTAMP
}
```

### Subscriptions Table
```sql
subscriptions {
  id: UUID PRIMARY KEY

  status: ENUM('active', 'inactive', 'trial')
  monthly_price: DECIMAL(10,2)

  -- Settings (JSON)
  config: JSON

  -- Billing

  closed_at: TIMESTAMP
  created_at: TIMESTAMP
}
```

exchange_accounts {

  status: ENUM('active', 'inactive', 'error')

  -- Balance

  equity: DECIMAL(18,8)

  created_at: TIMESTAMP
